import { useRef, useEffect } from 'react'
import { Chart, BinanceAdapter, DARK_THEME } from '@tradecanvas/chart'

export function TradingChart() {
  const containerRef = useRef<HTMLDivElement>(null)
  const chartRef = useRef<Chart | null>(null)

  useEffect(() => {
    if (!containerRef.current) return

    const chart = new Chart(containerRef.current, {
      chartType: 'candlestick',
      theme: DARK_THEME,
      autoScale: true,
      features: { drawings: true, indicators: true, volume: true },

    return () => chart.destroy()

}

Data Format

All chart data uses the OHLCBar interface.

OHLCBar Interface
interface OHLCBar {
  time: number    // Unix timestamp in milliseconds
  open: number    // Opening price
  high: number    // Highest price in the period
  low: number     // Lowest price in the period
  close: number   // Closing price
  volume: number  // Trading volume
}

Load an array of bars directly. The array must be sorted by time ascending.

Static data
const bars = await fetch('/api/bars').then(r => r.json())

Connect to a live data source using an adapter. The chart handles history loading, real-time updates, and auto-reconnection.

Custom Data Adapter

Build your own data source by implementing the DataAdapter interface.

DataAdapter Interface
interface DataAdapter {
  readonly name: string

  connect(config: DataAdapterConfig): void
  disconnect(): void
  getConnectionState(): ConnectionState

  fetchHistory(
    symbol: string,
    timeframe: TimeFrame,
    limit?: number,
  ): Promise<OHLCBar[]>

  on(event: DataAdapterEventType, listener: Function): void
  off(event: DataAdapterEventType, listener: Function): void
  dispose(): void
}

Example: REST + WebSocket Adapter

CustomAdapter.ts
import type {
  DataAdapter, DataAdapterConfig, DataAdapterEventType,
  ConnectionState, OHLCBar, TimeFrame,
} from '@tradecanvas/chart'

export class CustomAdapter implements DataAdapter {
  readonly name = 'custom'
  private ws: WebSocket | null = null
  private state: ConnectionState = 'disconnected'
  private listeners = new Map<string, Set<Function>>()

  connect(config: DataAdapterConfig): void {
    this.state = 'connecting'
    this.emit('connectionChange', { state: this.state })

    this.ws = new WebSocket(`wss://your-api.com/ws/${config.symbol}`)

    this.ws.onopen = () => {
      this.state = 'connected'
      this.emit('connectionChange', { state: 'connected' })
    }

    this.ws.onmessage = (e) => {
      const tick = JSON.parse(e.data)
      this.emit('tick', {
        time: tick.timestamp,
        price: tick.price,
        volume: tick.volume,
      })
    }

    this.ws.onerror = () => {
      this.state = 'error'
      this.emit('connectionChange', { state: 'error' })
    }
  }

  disconnect(): void {
    this.ws?.close()
    this.state = 'disconnected'
  }

  getConnectionState(): ConnectionState {
    return this.state
  }

  async fetchHistory(
    symbol: string,
    timeframe: TimeFrame,
    limit = 500,
  ): Promise<OHLCBar[]> {
    const res = await fetch(
      `https://your-api.com/bars?symbol=${symbol}&tf=${timeframe}&limit=${limit}`
    )
    return res.json()
  }

  on(event: DataAdapterEventType, listener: Function): void {
    if (!this.listeners.has(event)) this.listeners.set(event, new Set())
    this.listeners.get(event)!.add(listener)
  }

  off(event: DataAdapterEventType, listener: Function): void {
    this.listeners.get(event)?.delete(listener)
  }

  dispose(): void {
    this.disconnect()
    this.listeners.clear()
  }

  private emit(event: string, data: unknown): void {
    this.listeners.get(event)?.forEach(fn =>
      fn({ type: event, data, timestamp: Date.now() })
    )
  }
}

Save / Load / Export

Persist chart state and export data in multiple formats.

State management
// ── Save to JSON string ──
const json = chart.saveState()
// Contains: drawings, indicators, theme, chartType, viewport

// ── Load from JSON string ──
chart.loadState(json)

// ── Save to localStorage ──
chart.saveState('my-chart-key')      // saves to localStorage
chart.loadStateFromStorage('my-chart-key') // loads from localStorage

// ── Download / upload file ──
chart.downloadState()                  // triggers .json file download
chart.downloadState('my-chart.json')   // custom filename
await chart.loadStateFromFile()         // opens file picker dialog

// ── Auto-save ──
chart.setAutoSave('chart-state', 5000)  // auto-save every 5s
chart.disableAutoSave()

// ── Data export ──
chart.exportVisibleData('csv')          // exports visible bars as CSV
chart.exportVisibleData('json')         // exports visible bars as JSON
chart.exportAllData('csv')              // exports all bars as CSV
chart.exportAllData('json')             // exports all bars as JSON
